function renderMasterNoticeBoard() { const container = document.getElementById('notice-board-target'); if (!container) return; container.innerHTML = ""; // 1. Sort notices to ensure urgent ones appear at the absolute top const sortedNotices = [...CMCC_NOTICES].sort((a, b) => { if (a.type === 'urgent' && b.type !== 'urgent') return -1; if (a.type !== 'urgent' && b.type === 'urgent') return 1; return 0; }); // 2. Loop through and dynamically generate clean, neutral UI blocks sortedNotices.forEach(notice => { const isUrgent = notice.type === 'urgent'; const noticeCard = document.createElement('div'); noticeCard.className = `p-4 mb-3 rounded-xl border transition-all duration-200 ${ isUrgent ? 'bg-red-50/60 border-red-200/80 shadow-sm' : 'bg-zinc-50/50 border-zinc-200 hover:bg-zinc-50' }`; // Generate attachment button layout if a PDF or download link is present const downloadButton = notice.attachment ? ` DOWNLOAD ATTACHMENT ` : ''; noticeCard.innerHTML = `
${notice.date} ${notice.type}

${notice.title}

${notice.content}

${downloadButton}
`; container.appendChild(noticeCard); }); }
